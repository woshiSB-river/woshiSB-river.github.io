T3

鲁迅说过：“看到异或，就要想到 trietrietrie ”。所以直接将所以数变为二进制放到 trietrietrie 中。

然后我们来考虑一个数能够对答案造成多大的贡献。

一个数在 trietrietrie 上，当他在一个节点，且下一个假设是走左节点（也就是 000 ），所以他这时会与当前节点的右子树中的所有数的异或后末尾的 000 为当前深度的值，统计答案即可。

上菜

#include<bits/stdc++.h>
#define mod 199907210507ll
#define int long long
inline int read()
{	int x=0;
	bool f=0;
	char c=getchar();
	while(!isdigit(c))f|=(c=='-'),c=getchar();
	while(isdigit(c))x=x*10+(c&15),c=getchar();
	return f?-x:x;
}
using namespace std;
int n,a[100005],T[6000005][2],cnt=1,num[6000005],siz[6000005],ans;
void fj(int x)
{	int now=1;
	siz[1]++;
	for(register int i=0;i<60;++i)
	{	int ls=x%2;
		if(!T[now][ls])T[now][ls]=++cnt;
		
		now=T[now][ls];
		siz[now]++;
		x/=2;
	}
}
int tongji(int x)
{	int now=0;
	for(register int i=0,p=1;i<60;++i)
	{	now=(now+(1LL<<i)%mod*siz[T[p][(x&1)^1]]%mod)%mod;
		p=T[p][x&1];x/=2;
	}
	return now;
}
signed main()
{	n=read();
	for(register int i=1;i<=n;++i)a[i]=read(),fj(a[i]);
	for(register int i=1;i<=n;++i)ans=(ans+tongji(a[i]))%mod;
	cout<<ans;
	return 0;
}
