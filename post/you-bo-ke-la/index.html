T1

第一眼看到此题为树形结构，于是想到了树形DP。

看到要求:

  • 祖先在儿子前
  • 同一父亲的儿子，要按照从左到右的顺序出现

所以设 fif_ifi​ 为以 iii 为根的子树的方案数。

由于要儿子要按照从左到右的顺序，但在 DPDPDP 时不好处理。所以考虑倒序的形式。

对于一个新加的子树，首先他的根肯定是放在整个序列的首位（因为条件2），然后他的儿子没有与其他子树的限制条件，所以可以随便放（即插空），插孔的方案数为 CL+siz−1siz−1{C_{L+siz-1}^{siz-1}}CL+siz−1siz−1​ （ LLL 为未加入新子树前的序列长度， sizsizsiz 为新加入子树的子树大小），然后用插空的方案数乘上新加入子树的方案数即为加入后的方案数。

fi=∏j∈son⁡(i)fj×CL+siz−1siz−1f_i=\prod_{j \in \operatorname{son}(i)}{f_j\times C_{L+siz-1}^{siz-1}} fi​=j∈son(i)∏​fj​×CL+siz−1siz−1​

#define int long long
#define mod 10007
inline int read()
{	int x=0;
	bool f=0;
	char c=getchar();
	while(!isdigit(c))f|=(c=='-'),c=getchar();
	while(isdigit(c))x=x*10+(c&15),c=getchar();
	return f?-x:x;
}
using namespace std;
int T,n,siz[1005],dp[1005],jc[10005];
inline int ksm(int x,int y)
{	int res=1;
	while(y)
	{	if(y&1)res=res*x%mod;
		x=x*x%mod;
		y>>=1;
	} 
	return res;
}
vector<int> Tr[1005];
inline int C(int s,int x)
{	return jc[x]*ksm(jc[x-s],mod-2)%mod*ksm(jc[s],mod-2)%mod;
}
void get_siz(int x)
{	siz[x]=1;
	int len=Tr[x].size();
	for(register int i=0;i<len;++i)
	{	get_siz(Tr[x][i]);
		siz[x]+=siz[Tr[x][i]]; 
	}
	return ;
}
void DP(int x)
{	dp[x]=1;
	int len=Tr[x].size();
	if(len==0)return ;
	DP(Tr[x][len-1]);
	int now=dp[Tr[x][len-1]],sum=siz[Tr[x][len-1]];
	for(register int i=len-2;i>=0;--i)
	{	int y=Tr[x][i];
		DP(y);
		now=now*dp[y]%mod*C(siz[y]-1,sum+siz[y]-1)%mod;
		sum+=siz[y];
	}
	dp[x]=dp[x]*now%mod;
	return ;
}
signed main()
{	freopen("t1.in","r",stdin);
	freopen("t1.out","w",stdout);
	T=read();
	jc[0]=1;
	for(register int i=1;i<=10000;++i)jc[i]=jc[i-1]*i%mod;
	while(T--)
	{	n=read();
		for(register int i=1;i<=n;++i)
		{	Tr[i].clear();
			int l=read();
			for(register int j=1;j<=l;++j)
			{	int x=read();
				Tr[i].push_back(x);
			}
		}
		get_siz(1);
		DP(1);
		printf("%lld\n",dp[1]);
	}
	return 0;
}

T2

看到让你求一个 a∼ba\sim ba∼b 的和，直接想到用前缀和优化一下，即求 1∼(a−1)1\sim (a-1)1∼(a−1) 和 1∼b1\sim b1∼b 的即可。

提供两种做法：

  • 看每一个 kkk ，求能开 kkk 次根的数的个数。
  • 将 bbb 开立方，枚举 1∼b31\sim \sqrt[3]{b}1∼3b​ 的个数统计即可。

#include<stdio.h>

long long f[71];

long long mi(long long a,long long b)
{
	if(b==0)
		return 1;
	long long ans=mi(a,b>>1);
	ans*=ans;
	if(b&1)
		ans=ans*a;
	return ans;
}

{
	long long a,b,i,j,k;
	freopen("t2.in","r",stdin);

	{
		scanf("%I64d%I64d",&a,&b);
		if(a==0)
			break;
		for(i=64;i>=1;--i)
		{
			long long qb=floor(pow(b,1/(double)i)+1e-10),qa=ceil(pow(a,1/(double)i)-1e-10);
			if(i<=5)
			{
				for(qb=qb+2;mi(qb,i)>b;--qb);
				for(qa=qa-2>=0?qa-2:0;mi(qa,i)<a;++qa);

			for(j=2;i*j<=64;++j)
				f[i]-=f[i*j];
		}
		long long ans=0;
		for(i=1;i<=64;++i)
			ans+=f[i]*i;
		printf("%I64d\n",ans);
	}
	return 0;
}

#define int long long
inline int read()
{	int x=0;
	bool f=0;
	char c=getchar();
	while(!isdigit(c))f|=(c=='-'),c=getchar();
	while(isdigit(c))x=x*10+(c&15),c=getchar();
	return f?-x:x;
}
using namespace std;
int a,b,bj[1000005];
inline int erfen(int x)
{	int l=0,r=1000005,ans=0;
	while(l<r)
	{	int mid=(l+r)>>1;
		if(mid*mid*mid<=x)l=mid+1,ans=mid;
		else r=mid-1;
	}
	if((ans+1)*(ans+1)*(ans+1)<=x)return ans+1;
	return ans;
}
inline int get_ans(int x)
{	int maxn=erfen(x),sum=0,gs=0,maxnn=sqrt(x);
	int hou=maxnn-maxn;
	for(register int i=2;i<=maxn;++i)
	{	if(bj[i])continue;
		int now=i,tp=1;
		while(now<=x)
		{	gs++;
			sum+=tp;
			if(now<=1000003)bj[now]=1;
			if(now>maxn&&now<=maxnn)hou--;
			if(x/now<i)break;
			now=now*i;
			tp+=1;
		}
	}
	return sum+(x-gs)+hou;
}
signed main()
{	freopen("t2.in","r",stdin);
	freopen("t2.out","w",stdout);
	while(1)
	{	a=read(),b=read();
		if(a+b==0)break;
		memset(bj,0,sizeof(bj));
		int la=get_ans(a-1);
		memset(bj,0,sizeof(bj));
		int lb=get_ans(b);
		printf("%lld\n",lb-la);
	}
	return 0;
}

vector<int> jh(vector<int> a,vector<int> b)
{	vector<int> c;
	int len=max(a.size(),b.size()),la=a.size(),lb=b.size();
	for(register int i=0;i<len;++i)
	{	int shuz=0;
		if(i<la)shuz+=a[i];
		if(i<lb)shuz+=b[i];
		c.push_back(shuz);
	}
	for(register int i=0;i<len;++i)
	{	if(c[i]>=10)
		{	if(i<len-1)c[i]-=10,c[i+1]+=1;
			else c[i]-=10,c.push_back(1);	
		}
	}
	return c;
}
